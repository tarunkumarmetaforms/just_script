FUNCTION CALCULATE_LOYALTY_SCORE(visit_frequency, recommendation_score) frequency_score = SWITCH(visit_frequency)

CASE 1: RETURN 5 // Daily

CASE 2: RETURN 4 // Several times a week CASE 3: RETURN 3 // Once a week

CASE 4: RETURN 2 // A few times a month CASE 5: RETURN 1 // Less often

CASE 6: RETURN 0 // First visit END SWITCH

loyalty_score = (frequency_score * 2) + (recommendation_score / 2) RETURN ROUND(loyalty_score, 1)

END FUNCTION
